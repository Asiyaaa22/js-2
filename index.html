function  factorial (n){
    let fact = 1;
    for(let i=n;i>=1;i--){
        fact*=i
    }
    return (fact);
}

let printNumber =(time, num) =>{
    return new Promise ((resolve, reject)=>{
        if (num){
            setTimeout(()=>{
                resolve(num());
            },time)
        }
        else{
            reject(console.log("failed"))
        }
    })
}
